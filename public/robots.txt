    <script>
        document.addEventListener('DOMContentLoaded', async () => {
            const container = document.getElementById('patterns-container');
            const footer = document.querySelector('footer p');

            try {
                const response = await fetch('http://localhost:1337/api/patterns');
                const data = await response.json();

                if (data.data && data.data.length > 0) {
                    footer.textContent = ''; // Rimuove il messaggio "Loading patterns..."

                    data.data.forEach(pattern => {
                        const card = document.createElement('div');
                        card.className = 'card';

                        const patternName = pattern.attributes?.PatternName;

                        if (patternName) {
                            card.textContent = patternName;
                        } else {
                            card.textContent = 'Unnamed pattern';
                        }

                        container.appendChild(card);
                    });
                } else {
                    footer.textContent = 'No patterns found.';
                }
            } catch (error) {
                console.error('Error fetching patterns:', error);
                footer.textContent = 'Failed to load patterns. Please try again later.';
            }
        });
    </script>
